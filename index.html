'leaning-left' : 'leaning-right'; enemy.classList.add(direction); console.log("IA Esquivou!"); // Remove a esquiva após 400ms (tempo para o jogador errar o golpe) setTimeout(() => { enemy.classList.remove('leaning-left', 'leaning-right'); aiState.isLeaning = false; }, 400); } }
